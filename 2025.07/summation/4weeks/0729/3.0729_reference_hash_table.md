```python
class HashTable:
    def __init__(self, size):
        self.size = size
        self.table = [None] * size

    def hash_func(self, key):
        return hash(key) % self.size

    def insert(self, key, value):
        index = self.hash_func(key)
        self.table[index] = value

    def get(self, key):
        index = self.hash_func(key)
        return self.table[index]
```
